        react: accent,
        active: accent,
        primary-accent: accent,
        primary-warn: warn,
        primary-border: primary,
        primary-text: (from: primary, method: 50%),
        primary-link: primary-text,
        primary-react: primary-link,
        primary-active: primary-react,
        accent-warn: warn,
        accent-border: accent,
        accent-text: text,
        accent-link: link,
        accent-react: react,
        accent-active: active,
        warn-border: warn,
        warn-text: text,
        warn-link: link,
        warn-react: react,
        warn-active: active
) !default;

    Description

    required: primary, field, text
